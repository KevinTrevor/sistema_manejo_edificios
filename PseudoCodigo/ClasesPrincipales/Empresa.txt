          Metodo Edificio()
          edificios<- Nuevo(ListaEnlazada())
          encargados <- Nuevo(ListaEnlazada())
          arrendatarios <- Nuevo(ListaEnlazada())
          FinMetodo


          Metodo mostrarEdificios():cadena
          edificioStr:cadena
          edificio: Nodo de Edificio
          Mientras(edificio <> nulo)Hacer
            edificioStr <- edificioStr + getEncargadoPorCedula(edificio.getinfo().getCedulaEncargado()) + edificio.getinfo().mostrar()
            edificio <- edificio.getSiguiente()
          Finmientras
          mostrarEdificios<- edificioStr
          FinMetodo

          Metodo mostrarLocalPorCodigo(cadena:codigo,entero:numeroPiso,cadena:codigoLocal)
          localStr:cadena
          edificioBuscado: Edificio
          edificioBuscado <- getEdificios().buscarDato(codigo)
          Si(edificioBuscado <> nulo)Entonces
            localstr <- getArrendadoPorCedula(edificioBuscado.getCedulaArrendatarioLocal(numeroPiso,codigoLocal)) + edificioBuscado.mostrarLocalPorCodigo(numeroPiso,codigoLocal)

          Finsi
          mostrarLocalPorCodigo <- localstr
          FinMetodo


          Metodo mostrarEdificioPorCodigo(cadena:codigo)
          edificioStr:cadena
          edificio:Edificio
          edificio <- getEdificios().buscarDato(codigo)
          Si(edificio <> nulo)Entonces
            edificioStr <-  getEncargadoPorCedula(edificio.getCedulaEncargado()) + edificio.mostrar()
          Finsi
          FinMetodo

          Metodo modificarEncargado(cadena:cedula,cadena:nombre,cadena:apellido,cadena:correo,cadena:telefono,cadena:telefonoMovil,LocalDate fechaContrato)
          encargadoBuscado: Encargado
          encargadoBuscado <- getEncargados().buscarDato(cedula)
          Si(encargadoBuscado <> nulo)Entonces
            encargadoBuscado.modificarEncargado(nombre,apellido,correo,telefono,telefonoMovil,fechaContrato)
            getEncargados().modificarDato(cedula,encargadoBuscado)
          Finsi
          FinMetodo

          Metodo modificarArrendatario(cadena:cedula,cadena:nombre,cadena:apellido,cadena:correo,cadena:telefono,LocalDate FechaAsignacion)
          arrendatarioBuscado: Arrendatario
          arrendatarioBuscado <- getArrendatarios().buscarDato(cedula)
          Si(arrendatarioBuscado <> nulo)Entonces
            arrendatarioBuscado.modificarArrendatario(nombre,apellido,correo,telefono,FechaAsignacion)
            getArrendatarios().modificarDato(cedula,arrendatarioBuscado)
          Finsi
          
          FinMetodo

          edificioBuscado: Edificio
          edificioBuscado <- getEdificios().buscarDato(codigo)
          Si(edificioBuscado <> nulo)Entonces
            edificioBuscado.modificarLocal(numeroPiso,codigoLocal,cedulaArrendatario,montoMensualidad)
            getEdificios().modificarDato(codigo,edificioBuscado)
          Finsi

          Metodo modificarPiso(cadena:codigo,entero:numeroPiso,cadena:cedulaEncargado,LocalDate: fechaRegistro)
          edificioBuscado:Edificio

            edificioBuscado.modificarPiso(numeroPiso,cedulaEncargado,fechaRegistro)

          Metodo modificarEdificio(cadena:codigo,cadena:identificacion,cadena:estado,cadena:direccion,cadena:cedulaEncargado,LocalDate: _fechaCreacion)
          edificioBuscado:Edificio
          edificioBuscado <- getEdificios().buscarDato(codigoLocal)
          Si(edificioBuscado <> nulo)Entonces
            edificioBuscado.modificarEdificio(identificacion,estado,direccion,cedulaEncargado,fechaCreacion)
            getEdificios().modificarDato(codigo,edificioBuscado)
          Finsi
          FinMetodo

          getEdificios().eliminarDato(codigo)

          Metodo eliminarPiso(cadena:codigo,entero:numeroPiso)
          edificioBuscado:Edificio
          edificio <- getEdificios().buscarDato(codigo)
          Si(edificioBuscado <> nulo)Entonces
            edificioBuscado.eliminarPiso(numeroPiso)
            getEdificios().modificarDato(codigo,edificioBuscado)
          Finsi
          FinMetodo

          Metodo agregarEdificios(Edificio: nuevoEdificio)
          nuevo:Nodo de Edificio
          nuevo <- nuevoEdificio
          listaEdificios: ListaEnlazada de Edificio
          listaEdificios <- getEdificios()

          Si(listaEdificios.estaVacio())Entonces
            listaEdificios.setInicio(nuevo)
            listaEdificios.setFin(nuevo)
            listaEdificios.setSize(listaEdificios.getSize()  + 1)
          Sino
            auxiliar: Nodo de Edificio
            auxiliar <- listaEdificios.getInicio()
            Si(nuevoEdificio.getCodigoLocal().compareTo(auxiliar.getInfo().getCodigo()) < 0)Entonces
              nuevo.setSiguiente(auxiliar)
              listaEdificios.setInicio(nuevo)
              listaEdificios.setSize(listaEdificios.getSize() + 1)
            Sino
              Mientras(nuevoEdificio.getCodigo().compareTo(auxiliar.getinfo().getCodigo())>0 && auxiliar.getSiguiente() <> nulo)Hacer
                auxiliar <- auxiliar.getSiguiente()
              Finmientras
              Si(auxiliar.getinfo().getCodigo().equals(nuevoEdificio.getCodigo()) = verdaero)Entonces
                Escribir "Ya existe un edificio"
              Sino
                nuevo.setSiguiente(auxiliar.getSiguiente())
                auxiliar.setSiguiente(nuevo)
              Finsi
              listaEdificios.setSize(listaEdificios.getSize() + 1)
            Finsi 
          Finsi
          FinMetodo



          Metodo agregarEncargado(Encargado:nuevoEncargado)
